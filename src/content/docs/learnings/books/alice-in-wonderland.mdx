## Query 1 — File Discovery & Metadata Tagging

```
let
    Source = SharePoint.Files(
        "https://yourcompany.sharepoint.com/sites/CMGDecarbonisationUnit",
        [ApiVersion = 15]
    ),

    FilteredToIncoming = Table.SelectRows(
        Source,
        each Text.Contains([Folder Path], "CMG_GHG_KPIs/Incoming")
    ),

    // Rename columns with spaces to avoid reference issues
    RenameColumns = Table.RenameColumns(
        FilteredToIncoming,
        {
            {"Folder Path", "FolderPath"},
            {"Date Modified", "DateModified"},
            {"Date Created", "DateCreated"},
            {"Date accessed", "DateAccessed"}
        }
    ),

    // Keep only Excel files
    FilteredToExcel = Table.SelectRows(
        RenameColumns,
        each [Extension] = ".xlsx" or [Extension] = ".xls"
    ),

    // Split folder path into parts for dynamic extraction
    AddPathParts = Table.AddColumn(
        FilteredToExcel,
        "Path_Parts",
        each Text.Split([FolderPath], "/")
    ),

    AddIncomingIndex = Table.AddColumn(
        AddPathParts,
        "Incoming_Index",
        each List.PositionOf([Path_Parts], "Incoming")
    ),

    // Fleet = segment right after "Incoming"
    AddFleet = Table.AddColumn(
        AddIncomingIndex,
        "Fleet",
        each if [Incoming_Index] >= 0
             then [Path_Parts]{[Incoming_Index] + 1}
             else "Unknown"
    ),

    // Year_Folder = segment right after Fleet
    AddYearFolder = Table.AddColumn(
        AddFleet,
        "Year_Folder",
        each
            let
                YearText = [Path_Parts]{[Incoming_Index] + 2}
            in
                if Value.Is(Value.FromText(YearText), type number)
                then Number.FromText(YearText)
                else null
    ),

    // Data_Type from file name
    AddDataType = Table.AddColumn(
        AddYearFolder,
        "Data_Type",
        each if Text.Contains(Text.Upper([Name]), "ACT") then "ACT"
             else if Text.Length([Name]) > 3
                  and Text.Start([Name], 1) = "F"
                  and Text.Contains([Name], "_FY") then "Forecast"
             else "Unknown"
    ),

    // Forecast_Number = two digits after "F"
    AddForecastNumber = Table.AddColumn(
        AddDataType,
        "Forecast_Number",
        each if [Data_Type] = "Forecast"
             then Text.Middle([Name], 1, 2)
             else null
    ),

    // FY_Year extraction
    AddFYYear = Table.AddColumn(
        AddForecastNumber,
        "FY_Year",
        each if [Data_Type] = "Forecast"
             then
                let
                    FYPos = Text.PositionOf([Name], "_FY"),
                    YearText = Text.Middle([Name], FYPos + 3, 4)
                in
                    if Value.Is(Value.FromText(YearText), type number)
                    then Number.FromText(YearText)
                    else null
             else [Year_Folder]
    ),

    // Flag forecast files — only include if FY_Year = current year
    AddCurrentYear = Table.AddColumn(
        AddFYYear,
        "Current_Year",
        each Date.Year(DateTime.LocalNow())
    ),

    AddIncludeFlag = Table.AddColumn(
        AddCurrentYear,
        "Include_File",
        each if [Data_Type] = "ACT" then true
             else if [Data_Type] = "Forecast"
                  and [FY_Year] = [Current_Year] then true
             else false
    ),

    FilterIncluded = Table.SelectRows(
        AddIncludeFlag,
        each [Include_File] = true
    ),

    // Remove helper columns
    RemoveHelpers = Table.RemoveColumns(
        FilterIncluded,
        {"Path_Parts", "Incoming_Index", "Current_Year", "Include_File"}
    ),

    // Final column selection
    SelectColumns = Table.SelectColumns(
        RemoveHelpers,
        {"Content", "Name", "Extension", "DateModified",
         "DateCreated", "FolderPath", "Fleet",
         "Year_Folder", "Data_Type", "Forecast_Number", "FY_Year"}
    )

in
    SelectColumns
```

## Query 3 — Column Selection & Row Cleaning

```
let
    Source = SharePoint.Files(
        "https://yourcompany.sharepoint.com/sites/CMGDecarbonisationUnit",
        [ApiVersion = 15]
    ),

    FilteredToIncoming = Table.SelectRows(
        Source,
        each Text.Contains([Folder Path], "CMG_GHG_KPIs/Incoming")
    ),

    SampleFile = Table.SelectRows(
        FilteredToIncoming,
        each [Name] = "COSTAFY23ACT.xlsx"
    ),

    FileContent = SampleFile{0}[Content],

    OpenWorkbook = Excel.Workbook(FileContent, null, true),

    FirstSheet = OpenWorkbook{0}[Data],

    // Keep only columns we need
    KeepColumns = Table.SelectColumns(
        FirstSheet,
        {"Column3", "Column4", "Column5", "Column6",
         "Column8", "Column9", "Column10", "Column11",
         "Column12", "Column13", "Column14", "Column15",
         "Column16", "Column17", "Column18", "Column19"}
    ),

    // Remove rows where both Column3 and Column4 are null
    RemoveEmptyRows = Table.SelectRows(
        KeepColumns,
        each not ([Column3] = null and [Column4] = null)
    ),

    // Remove the days rows (Column3 = "# DAYS")
    RemoveDaysRows = Table.SelectRows(
        RemoveEmptyRows,
        each [Column3] <> "# DAYS"
    ),

    // Remove the month names row (Column8 = "DEC")
    RemoveMonthRow = Table.SelectRows(
        RemoveDaysRows,
        each [Column8] <> "DEC"
    ),

    // Add Parameter column
    // When Column5 = "SHIP CLASS" → this is a header row → take Column4 as parameter name
    // Otherwise null — we will fill down next
    AddParameter = Table.AddColumn(
        RemoveMonthRow,
        "Parameter",
        each if [Column5] = "SHIP CLASS"
             then [Column4]
             else null
    ),

    // Fill Parameter downward so every vessel row gets tagged
    FillParameter = Table.FillDown(
        AddParameter,
        {"Parameter"}
    ),

    // Now remove the block header rows
    // They are identified by Column5 = "SHIP CLASS"
    RemoveHeaderRows = Table.SelectRows(
        FillParameter,
        each [Column5] <> "SHIP CLASS"
    )

in
    RemoveHeaderRows
```

## Query 4 — Full Transform: Unpivot, Pivot & Metadata

```
let
    // =========================================================
    // METADATA
    // =========================================================
    Source = SharePoint.Files(
        "https://yourcompany.sharepoint.com/sites/CMGDecarbonisationUnit",
        [ApiVersion = 15]
    ),

    FilteredToIncoming = Table.SelectRows(
        Source,
        each Text.Contains([Folder Path], "CMG_GHG_KPIs/Incoming")
    ),

    RenameMetaCols = Table.RenameColumns(
        FilteredToIncoming,
        {
            {"Folder Path", "FolderPath"},
            {"Date modified", "DateModified"},
            {"Date created", "DateCreated"},
            {"Date accessed", "DateAccessed"}
        }
    ),

    SampleFile = Table.SelectRows(
        RenameMetaCols,
        each [Name] = "COSTAFY23ACT.xlsx"
    ),

    // Extract metadata values
    FileName    = SampleFile{0}[Name],
    FilePath    = SampleFile{0}[FolderPath],
    DateMod     = SampleFile{0}[DateModified],
    DateCre     = SampleFile{0}[DateCreated],
    DateAcc     = SampleFile{0}[DateAccessed],
    FileContent = SampleFile{0}[Content],

    // =========================================================
    // OPEN WORKBOOK
    // =========================================================
    OpenWorkbook = Excel.Workbook(FileContent, null, true),
    SheetName    = OpenWorkbook{0}[Name],
    FirstSheet   = OpenWorkbook{0}[Data],

    // =========================================================
    // EXTRACT MONTH NAMES FROM ROW 4
    // =========================================================
    MonthRow = Table.SelectRows(
        FirstSheet,
        each [Column8] = "DEC"
    ),

    MonthMapping = [
        Column8  = Record.Field(MonthRow{0}, "Column8"),
        Column9  = Record.Field(MonthRow{0}, "Column9"),
        Column10 = Record.Field(MonthRow{0}, "Column10"),
        Column11 = Record.Field(MonthRow{0}, "Column11"),
        Column12 = Record.Field(MonthRow{0}, "Column12"),
        Column13 = Record.Field(MonthRow{0}, "Column13"),
        Column14 = Record.Field(MonthRow{0}, "Column14"),
        Column15 = Record.Field(MonthRow{0}, "Column15"),
        Column16 = Record.Field(MonthRow{0}, "Column16"),
        Column17 = Record.Field(MonthRow{0}, "Column17"),
        Column18 = Record.Field(MonthRow{0}, "Column18"),
        Column19 = Record.Field(MonthRow{0}, "Column19")
    ],

    // =========================================================
    // KEEP ONLY RELEVANT COLUMNS
    // =========================================================
    KeepColumns = Table.SelectColumns(
        FirstSheet,
        {"Column1", "Column3", "Column4", "Column5", "Column6",
         "Column8", "Column9", "Column10", "Column11",
         "Column12", "Column13", "Column14", "Column15",
         "Column16", "Column17", "Column18", "Column19"}
    ),

    // =========================================================
    // REMOVE UNWANTED ROWS
    // =========================================================
    RemoveMonthRow = Table.SelectRows(
        KeepColumns,
        each [Column8] <> "DEC"
    ),

    RemoveDaysRows = Table.SelectRows(
        RemoveMonthRow,
        each [Column3] <> "# DAYS"
    ),

    RemoveCheckRows = Table.SelectRows(
        RemoveDaysRows,
        each [Column1] <> "Check"
    ),

    RemoveEmptyRows = Table.SelectRows(
        RemoveCheckRows,
        each not ([Column3] = null and [Column4] = null)
    ),

    // =========================================================
    // ADD PARAMETER COLUMN
    // Block header row detected by Column5 = "SHIP CLASS"
    // Column4 in that row = parameter name
    // Fill down to all vessel rows below
    // =========================================================
    AddParameter = Table.AddColumn(
        RemoveEmptyRows,
        "Parameter",
        each if [Column5] = "SHIP CLASS"
             then Text.From([Column4])
             else null
    ),

    FillParameter = Table.FillDown(
        AddParameter,
        {"Parameter"}
    ),

    // =========================================================
    // EXTRACT SHIP CLASS AND ALB
    // Only from ALBDs FULL MONTH block
    // =========================================================
    AddShipClass = Table.AddColumn(
        FillParameter,
        "Ship_Class",
        each if [Parameter] = "ALBDs FULL MONTH"
                and [Column5] <> "SHIP CLASS"
                and [Column5] <> null
                and Text.From([Column5]) <> "0"
             then Text.From([Column5])
             else null
    ),

    AddALB = Table.AddColumn(
        AddShipClass,
        "ALB",
        each if [Parameter] = "ALBDs FULL MONTH"
                and [Column6] <> null
                and [Column6] <> 0
             then [Column6]
             else null
    ),

    FillShipALB = Table.FillDown(
        AddALB,
        {"Ship_Class", "ALB"}
    ),

    // =========================================================
    // KEEP ONLY VALID VESSEL ROWS
    // =========================================================
    KeepVesselRows = Table.SelectRows(
        FillShipALB,
        each
            // Must have a text vessel name in Column4
            [Column4] <> null
            and [Column4] <> 0
            and Text.From([Column4]) <> "0"
            and Text.From([Column4]) <> "0.00"
            // Must not be a block header row
            and [Column5] <> "SHIP CLASS"
            // Must not be a fleet total row
            // Fleet total rows have purely numeric Column3
            // Vessel rows have alphanumeric like LO346
            and not (
                Text.Length(
                    Text.Remove(
                        Text.From([Column3]),
                        {"0".."9"}
                    )
                ) = 0
            )
    ),

    // =========================================================
    // RENAME COLUMNS
    // =========================================================
    RenameMonthCols = Table.RenameColumns(
        KeepVesselRows,
        {
            {"Column4",  "Vessel_Name"},
            {"Column3",  "Vessel_Code"},
            {"Column8",  MonthMapping[Column8]},
            {"Column9",  MonthMapping[Column9]},
            {"Column10", MonthMapping[Column10]},
            {"Column11", MonthMapping[Column11]},
            {"Column12", MonthMapping[Column12]},
            {"Column13", MonthMapping[Column13]},
            {"Column14", MonthMapping[Column14]},
            {"Column15", MonthMapping[Column15]},
            {"Column16", MonthMapping[Column16]},
            {"Column17", MonthMapping[Column17]},
            {"Column18", MonthMapping[Column18]},
            {"Column19", MonthMapping[Column19]}
        }
    ),

    // Drop columns no longer needed
    DropExtraCols = Table.RemoveColumns(
        RenameMonthCols,
        {"Column1", "Column5", "Column6", "Vessel_Code"}
    ),

    // =========================================================
    // UNPIVOT MONTH COLUMNS
    // =========================================================
    UnpivotMonths = Table.UnpivotOtherColumns(
        DropExtraCols,
        {"Vessel_Name", "Ship_Class", "ALB", "Parameter"},
        "Month",
        "Value"
    ),

    // =========================================================
    // PIVOT PARAMETERS INTO COLUMNS
    // =========================================================
    PivotParameters = Table.Pivot(
        UnpivotMonths,
        List.Distinct(UnpivotMonths[Parameter]),
        "Parameter",
        "Value",
        List.Sum
    ),

    // =========================================================
    // ADD CALENDAR YEAR AND MONTH NUMBER
    // FY runs DEC(year-1) to NOV(year)
    // =========================================================
    FYYear = 2023,

    AddCalendarYear = Table.AddColumn(
        PivotParameters,
        "Calendar_Year",
        each if [Month] = "DEC"
             then FYYear - 1
             else FYYear
    ),

    AddMonthNumber = Table.AddColumn(
        AddCalendarYear,
        "Month_Number",
        each if [Month] = "JAN" then 1
             else if [Month] = "FEB" then 2
             else if [Month] = "MAR" then 3
             else if [Month] = "APR" then 4
             else if [Month] = "MAY" then 5
             else if [Month] = "JUN" then 6
             else if [Month] = "JUL" then 7
             else if [Month] = "AUG" then 8
             else if [Month] = "SEP" then 9
             else if [Month] = "OCT" then 10
             else if [Month] = "NOV" then 11
             else if [Month] = "DEC" then 12
             else null
    ),

    // =========================================================
    // ADD ALL METADATA COLUMNS
    // =========================================================
    AddMeta1  = Table.AddColumn(AddMonthNumber, "Source_File",   each FileName),
    AddMeta2  = Table.AddColumn(AddMeta1,       "Source_Path",   each FilePath),
    AddMeta3  = Table.AddColumn(AddMeta2,       "Sheet_Name",    each SheetName),
    AddMeta4  = Table.AddColumn(AddMeta3,       "Date_Modified", each DateMod),
    AddMeta5  = Table.AddColumn(AddMeta4,       "Date_Created",  each DateCre),
    AddMeta6  = Table.AddColumn(AddMeta5,       "Date_Accessed", each DateAcc),
    AddMeta7  = Table.AddColumn(AddMeta6,       "Fleet",         each "Costa"),
    AddMeta8  = Table.AddColumn(AddMeta7,       "Year_Folder",   each 2023),
    AddMeta9  = Table.AddColumn(AddMeta8,       "Data_Type",     each "ACT"),
    AddMeta10 = Table.AddColumn(AddMeta9,       "FY_Year",       each FYYear),

    // =========================================================
    // FINAL COLUMN ORDERING
    // =========================================================
    ReorderColumns = Table.ReorderColumns(
        AddMeta10,
        {
            "Source_File", "Source_Path", "Sheet_Name",
            "Date_Modified", "Date_Created", "Date_Accessed",
            "Fleet", "Year_Folder", "Data_Type", "FY_Year",
            "Vessel_Name", "Ship_Class", "ALB",
            "Calendar_Year", "Month", "Month_Number"
        }
        & List.Difference(
            Table.ColumnNames(AddMeta10),
            {
                "Source_File", "Source_Path", "Sheet_Name",
                "Date_Modified", "Date_Created", "Date_Accessed",
                "Fleet", "Year_Folder", "Data_Type", "FY_Year",
                "Vessel_Name", "Ship_Class", "ALB",
                "Calendar_Year", "Month", "Month_Number"
            }
        )
    )

in
    ReorderColumns

    ),

        each if [Incoming_Index] >= 0
             then [Path_Parts]{[Incoming_Index] + 1}
             else "Unknown"
    ),

    // =========================================================
    // EXTRACT YEAR FOLDER
    // Year = folder name immediately after Fleet
    // =========================================================
    AddYearFolder = Table.AddColumn(
        AddFleet,
        "Year_Folder",
        each
            let
                YearText = [Path_Parts]{[Incoming_Index] + 2}
            in
                if Value.Is(Value.FromText(YearText), type number)
                then Number.FromText(YearText)
                else null
    ),

    // =========================================================
    // EXTRACT DATA TYPE
    // ACT files contain "ACT" in name
    // Forecast files start with F and contain "_FY"
    // =========================================================
    AddDataType = Table.AddColumn(
        AddYearFolder,
        "Data_Type",
        each if Text.Contains(Text.Upper([Name]), "ACT") then "ACT"
             else if Text.Start([Name], 1) = "F"
                  and Text.Contains([Name], "_FY") then "Forecast"
             else "Unknown"
    ),

    // Two digits after F e.g. F05_FY2026 = 05
    // =========================================================
    AddForecastNumber = Table.AddColumn(

        each if [Data_Type] = "Forecast"
             then Text.Middle([Name], 1, 2)
             else null
    ),

## Query 9 — GHG\_KPI\_Combined (Step 3: Final Output)

```
let
    // =========================================================
    // GET FILE LIST
    // =========================================================
    FileList = File_List,

    // =========================================================
    // CONVERT TO LIST OF RECORDS
    // Each record = one file row passed to function
    // =========================================================
    FileRecords = Table.ToRecords(FileList),

    // =========================================================
    // APPLY FUNCTION TO EACH FILE
    // Each file processed individually
    // Results collected into a list of tables
    // =========================================================
    ProcessedTables = List.Transform(
        FileRecords,
        each fn_ProcessFile(_)
    ),

    // =========================================================
    // COMBINE ALL TABLES
    // Table.Combine handles different columns per file
    // Missing columns filled with null automatically
    // =========================================================
    CombinedTable = Table.Combine(ProcessedTables),

    // =========================================================
    // FINAL SORT
    // Sort by Fleet, FY_Year, Vessel_Name, Month_Number
    // =========================================================
    SortedTable = Table.Sort(
        CombinedTable,
        {
            {"Fleet",        Order.Ascending},
            {"FY_Year",      Order.Ascending},
            {"Vessel_Name",  Order.Ascending},
            {"Month_Number", Order.Ascending}
        }
    )

in
    SortedTable
