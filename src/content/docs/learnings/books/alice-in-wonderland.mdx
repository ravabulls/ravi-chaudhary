## Query 1 — File\_List

```
let
    // =========================================================
    // SINGLE SHAREPOINT CALL — METADATA ONLY FIRST
    // No content downloaded at this stage
    // =========================================================
    Source = SharePoint.Files(
        "https://yourcompany.sharepoint.com/sites/CMGDecarbonisationUnit",
        [ApiVersion = 15]
    ),

    // Filter to Incoming folder — metadata only, very lightweight
    FilteredToIncoming = Table.SelectRows(
        Source,
        each Text.Contains([Folder Path], "CMG_GHG_KPIs/Incoming")
    ),

    // Keep only Excel files
    FilteredToExcel = Table.SelectRows(
        FilteredToIncoming,
        each [Extension] = ".xlsx" or [Extension] = ".xls"
    ),

    // Rename columns with spaces
    RenameMetaCols = Table.RenameColumns(
        FilteredToExcel,
        {
            {"Folder Path", "FolderPath"},
            {"Date modified", "DateModified"},
            {"Date created", "DateCreated"}
        }
    ),

    // Fix date columns to proper datetime format
    FixDates = Table.TransformColumns(
        RenameMetaCols,
        {
            {"DateModified", each if _ = null then null
                else DateTime.From(_), type datetime},
            {"DateCreated",  each if _ = null then null
                else DateTime.From(_), type datetime}
        }
    ),

    // =========================================================
    // EXTRACT FLEET DYNAMICALLY
    // Fleet = folder immediately after Incoming
    // Works for any new fleet added in future
    // =========================================================
    AddPathParts = Table.AddColumn(
        FixDates,
        "Path_Parts",
        each Text.Split([FolderPath], "/")
    ),

    AddIncomingIndex = Table.AddColumn(
        AddPathParts,
        "Incoming_Index",
        each List.PositionOf([Path_Parts], "Incoming")
    ),

    AddFleet = Table.AddColumn(
        AddIncomingIndex,
        "Fleet",
        each if [Incoming_Index] >= 0
             then [Path_Parts]{[Incoming_Index] + 1}
             else "Unknown"
    ),

    // Year = folder immediately after Fleet
    AddYearFolder = Table.AddColumn(
        AddFleet,
        "Year_Folder",
        each
            let YearText = [Path_Parts]{[Incoming_Index] + 2}
            in  if Value.Is(Value.FromText(YearText), type number)
                then Number.FromText(YearText)
                else null
    ),

    // =========================================================
    // DATA TYPE DETECTION
    // ACT files contain ACT in name
    // Forecast files start with F and contain _FY
    // =========================================================
    AddDataType = Table.AddColumn(
        AddYearFolder,
        "Data_Type",
        each if Text.Contains(Text.Upper([Name]), "ACT") then "ACT"
             else if Text.Start([Name], 1) = "F"
                  and Text.Contains([Name], "_FY") then "Forecast"
             else "Unknown"
    ),

    AddForecastNumber = Table.AddColumn(
        AddDataType,
        "Forecast_Number",
        each if [Data_Type] = "Forecast"
             then Text.Middle([Name], 1, 2)
             else null
    ),

    AddFYYear = Table.AddColumn(
        AddForecastNumber,
        "FY_Year",
        each if [Data_Type] = "Forecast"
             then
                let
                    FYPos    = Text.PositionOf([Name], "_FY"),
                    YearText = Text.Middle([Name], FYPos + 3, 4)
                in
                    if Value.Is(Value.FromText(YearText), type number)
                    then Number.FromText(YearText)
                    else null
             else [Year_Folder]
    ),

    // =========================================================
    // FILE INCLUSION RULES
    // Current year folder → Forecast files only
    // Older year folders → ACT files only
    // =========================================================
    CurrentYear = Date.Year(DateTime.LocalNow()),

    AddIncludeFlag = Table.AddColumn(
        AddFYYear,
        "Include_File",
        each
            if [Year_Folder] = CurrentYear
               and [Data_Type] = "Forecast" then true
            else if [Year_Folder] < CurrentYear
               and [Data_Type] = "ACT" then true
            else false
    ),

    // Apply filter BEFORE downloading any content
    // Only files that pass this filter get downloaded
    FilterIncluded = Table.SelectRows(
        AddIncludeFlag,
        each [Include_File] = true
    ),

    // Remove helper columns before buffering
    RemoveHelpers = Table.RemoveColumns(
        FilterIncluded,
        {
            "Path_Parts", "Incoming_Index",
            "Include_File", "Extension",
            "Attributes", "Date accessed",
            "FolderPath"
        }
    ),

    // =========================================================
    // BUFFER CONTENT ONLY FOR INCLUDED FILES
    // This is the ONLY point where files are downloaded
    // Correct placement: AFTER all filtering is done
    // =========================================================
    BufferContent = Table.TransformColumns(
        RemoveHelpers,
        {"Content", each Binary.Buffer(_)}
    ),

    // Buffer entire table into memory
    // Prevents any further SharePoint evaluation
    FinalBuffer = Table.Buffer(BufferContent)

in
    FinalBuffer
```

## Query 2 — cmg\_ghg\_master\_query

```
let
    // =========================================================
    // GET BUFFERED FILE LIST
    // All content already in memory — zero SharePoint calls below
    // =========================================================
    FileList = Table.Buffer(File_List),

    // Capture refresh time ONCE here outside the row loop
    // Keeping DateTime.LocalNow() here prevents privacy firewall
    // from triggering on every file row
    RefreshTime = DateTime.LocalNow(),

    // =========================================================
    // PROCESS EACH FILE INLINE
    // Table.AddColumn keeps everything in one evaluation context
    // No function calls = no context boundaries = no re-downloads
    // =========================================================
    AddProcessed = Table.AddColumn(
        FileList,
        "ProcessedData",
        each
            let
                // Local metadata for this file row
                FYYear_v    = [FY_Year],
                Fleet_v     = [Fleet],
                Year_v      = [Year_Folder],
                DataType_v  = [Data_Type],
                FcstNum_v   = [Forecast_Number],
                FileName_v  = [Name],
                DateMod_v   = [DateModified],
                DateCre_v   = [DateCreated],

                // Open workbook from buffered binary
                Wb          = Excel.Workbook([Content], null, true),
                SheetName_v = Wb{0}[Name],
                SheetData   = Wb{0}[Data],

                // Replace all cell errors with null
                CleanData   = Table.ReplaceErrorValues(
                    SheetData,
                    List.Transform(
                        Table.ColumnNames(SheetData),
                        each {_, null}
                    )
                ),

                // Buffer sheet into memory
                FirstSheet  = Table.Buffer(CleanData),

                // Extract month names from row where Column8 = DEC
                MonthRow    = Table.SelectRows(FirstSheet, each [Column8] = "DEC"),
                M1  = Record.Field(MonthRow{0}, "Column8"),
                M2  = Record.Field(MonthRow{0}, "Column9"),
                M3  = Record.Field(MonthRow{0}, "Column10"),
                M4  = Record.Field(MonthRow{0}, "Column11"),
                M5  = Record.Field(MonthRow{0}, "Column12"),
                M6  = Record.Field(MonthRow{0}, "Column13"),
                M7  = Record.Field(MonthRow{0}, "Column14"),
                M8  = Record.Field(MonthRow{0}, "Column15"),
                M9  = Record.Field(MonthRow{0}, "Column16"),
                M10 = Record.Field(MonthRow{0}, "Column17"),
                M11 = Record.Field(MonthRow{0}, "Column18"),
                M12 = Record.Field(MonthRow{0}, "Column19"),

                // Keep only relevant columns
                KeepCols = Table.SelectColumns(FirstSheet, {
                    "Column1", "Column3", "Column4",
                    "Column5", "Column6",
                    "Column8", "Column9", "Column10",
                    "Column11", "Column12", "Column13",
                    "Column14", "Column15", "Column16",
                    "Column17", "Column18", "Column19"
                }),

                // Remove unwanted rows
                R1 = Table.SelectRows(KeepCols,
                    each [Column8] <> "DEC"),
                R2 = Table.SelectRows(R1,
                    each [Column3] <> "# DAYS"),
                R3 = Table.SelectRows(R2,
                    each [Column1] <> "Check"),
                R4 = Table.SelectRows(R3,
                    each not ([Column3] = null and [Column4] = null)),

                // Add parameter column and fill down
                AddPar  = Table.AddColumn(R4, "Parameter",
                    each if [Column5] = "SHIP CLASS"
                         then Text.From([Column4])
                         else null),
                FillPar = Table.FillDown(AddPar, {"Parameter"}),

                // Extract Ship Class and ALB from ALBDs FULL MONTH only
                AddSC = Table.AddColumn(FillPar, "Ship_Class",
                    each if [Parameter] = "ALBDs FULL MONTH"
                            and [Column5] <> "SHIP CLASS"
                            and [Column5] <> null
                            and Text.From([Column5]) <> "0"
                         then Text.From([Column5])
                         else null),

                AddALB_col = Table.AddColumn(AddSC, "ALB",
                    each if [Parameter] = "ALBDs FULL MONTH"
                            and [Column6] <> null
                            and [Column6] <> 0
                         then [Column6]
                         else null),

                FillSALB = Table.FillDown(
                    AddALB_col, {"Ship_Class", "ALB"}),

                // Keep only valid vessel rows
                VesselRows = Table.SelectRows(FillSALB,
                    each
                        [Column4] <> null
                        and [Column4] <> 0
                        and Text.From([Column4]) <> "0"
                        and Text.From([Column4]) <> "0.00"
                        and [Column5] <> "SHIP CLASS"
                        and not (
                            Text.Length(
                                Text.Remove(
                                    Text.From([Column3]),
                                    {"0".."9"}
                                )
                            ) = 0
                        )
                ),

                // Rename columns
                Renamed = Table.RenameColumns(VesselRows, {
                    {"Column3", "Vessel_Code"},
                    {"Column4", "Vessel_Name"},
                    {"Column5", "SC_Raw"},
                    {"Column6", "ALB_Raw"},
                    {"Column8",  M1},  {"Column9",  M2},
                    {"Column10", M3},  {"Column11", M4},
                    {"Column12", M5},  {"Column13", M6},
                    {"Column14", M7},  {"Column15", M8},
                    {"Column16", M9},  {"Column17", M10},
                    {"Column18", M11}, {"Column19", M12}
                }),

                Dropped = Table.RemoveColumns(Renamed,
                    {"Column1", "Vessel_Code", "SC_Raw", "ALB_Raw"}),

                // Unpivot month columns into rows
                Unpivoted = Table.UnpivotOtherColumns(Dropped,
                    {"Vessel_Name", "Ship_Class", "ALB", "Parameter"},
                    "Month", "Value"),

                // Buffer before pivot
                Buf       = Table.Buffer(Unpivoted),
                ParamList = List.Buffer(List.Distinct(Buf[Parameter])),

                // Pivot parameters into columns
                // One vessel one month one row
                Pivoted = Table.Pivot(
                    Buf, ParamList, "Parameter", "Value", List.Sum),

                // Add calendar year
                WithYear = Table.AddColumn(Pivoted, "Calendar_Year",
                    each if [Month] = "DEC"
                         then FYYear_v - 1
                         else FYYear_v),

                // Add month number
                WithMNum = Table.AddColumn(WithYear, "Month_Number",
                    each if [Month] = "JAN" then 1
                         else if [Month] = "FEB" then 2
                         else if [Month] = "MAR" then 3
                         else if [Month] = "APR" then 4
                         else if [Month] = "MAY" then 5
                         else if [Month] = "JUN" then 6
                         else if [Month] = "JUL" then 7
                         else if [Month] = "AUG" then 8
                         else if [Month] = "SEP" then 9
                         else if [Month] = "OCT" then 10
                         else if [Month] = "NOV" then 11
                         else if [Month] = "DEC" then 12
                         else null),

                // Add all metadata columns
                A1  = Table.AddColumn(WithMNum, "Source_File",
                          each FileName_v),
                A2  = Table.AddColumn(A1,  "Sheet_Name",
                          each SheetName_v),
                A3  = Table.AddColumn(A2,  "Date_Modified",
                          each DateMod_v),
                A4  = Table.AddColumn(A3,  "Date_Created",
                          each DateCre_v),
                A5  = Table.AddColumn(A4,  "Refresh_DateTime",
                          each RefreshTime),
                A6  = Table.AddColumn(A5,  "Fleet",
                          each Fleet_v),
                A7  = Table.AddColumn(A6,  "Year_Folder",
                          each Year_v),
                A8  = Table.AddColumn(A7,  "Data_Type",
                          each DataType_v),
                A9  = Table.AddColumn(A8,  "Forecast_Number",
                          each FcstNum_v),
                A10 = Table.AddColumn(A9,  "FY_Year",
                          each FYYear_v),

                // Reorder columns — metadata first then data
                FixedCols = {
                    "Source_File", "Sheet_Name",
                    "Date_Modified", "Date_Created", "Refresh_DateTime",
                    "Fleet", "Year_Folder", "Data_Type", "Forecast_Number",
                    "FY_Year", "Vessel_Name", "Ship_Class", "ALB",
                    "Calendar_Year", "Month", "Month_Number"
                },

                Reordered = Table.ReorderColumns(A10,
                    FixedCols &
                    List.Difference(Table.ColumnNames(A10), FixedCols))

            in
                Reordered
    ),

    // =========================================================
    // COMBINE ALL PROCESSED TABLES
    // Table.Combine handles different columns per file
    // Missing parameters filled with null automatically
    // =========================================================
    CombinedTable = Table.Combine(AddProcessed[ProcessedData]),

    // Final sort
    SortedTable = Table.Sort(
        CombinedTable,
        {
            {"Fleet",        Order.Ascending},
            {"FY_Year",      Order.Ascending},
            {"Vessel_Name",  Order.Ascending},
            {"Month_Number", Order.Ascending}
        }
    )

in
    SortedTable

#### Calculated Column 1 — Year Version Label

```
Year_Version = 
IF(
    'cmg_ghg_master_query'[Data_Type] = "ACT",
